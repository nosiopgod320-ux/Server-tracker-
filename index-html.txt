<script>
  // Point this at your GitHub Pages data file
  const DATA_URL   = "./data/servers.json";
  const REFRESH_MS = 60_000;

  // Which seas each event belongs to
  const EVENT_SEA = {
    PirateRaid:     "Sea 3",
    FactoryRaid:    "Sea 2",
    FruitSpawn:     "All seas",
    LegendarySword: "Sea 2",
    FistOfDarkness: "Sea 2",
    GodChalice:     "Sea 3",
  };

  let activeEvent  = "PirateRaid";
  let activeSea    = "all";
  let serverData   = null;
  let tickInterval = null;

  // ── Sea filter ──────────────────────────────────────────────────────────
  document.querySelectorAll(".sea-btn").forEach(btn => {
    btn.addEventListener("click", () => {
      document.querySelectorAll(".sea-btn").forEach(b => b.classList.remove("active"));
      btn.classList.add("active");
      activeSea = btn.dataset.sea;
      updateTabVisibility();
      renderServers();
    });
  });

  function updateTabVisibility() {
    document.querySelectorAll(".tab").forEach(tab => {
      const eventSea = EVENT_SEA[tab.dataset.event] ?? "";
      const visible =
        activeSea === "all" ||
        eventSea === activeSea ||
        eventSea === "All seas";
      tab.style.display = visible ? "" : "none";
    });

    // If the active event tab is now hidden, switch to the first visible one
    const activeTab = document.querySelector(`.tab[data-event="${activeEvent}"]`);
    if (activeTab && activeTab.style.display === "none") {
      const firstVisible = document.querySelector(".tab:not([style*='none'])");
      if (firstVisible) {
        document.querySelectorAll(".tab").forEach(t => t.classList.remove("active"));
        firstVisible.classList.add("active");
        activeEvent = firstVisible.dataset.event;
      }
    }
  }

  // ── Event tabs ───────────────────────────────────────────────────────────
  document.querySelectorAll(".tab").forEach(btn => {
    btn.addEventListener("click", () => {
      document.querySelectorAll(".tab").forEach(t => t.classList.remove("active"));
      btn.classList.add("active");
      activeEvent = btn.dataset.event;
      renderServers();
    });
  });

  // ── Fetch ─────────────────────────────────────────────────────────────────
  async function fetchData() {
    try {
      const res = await fetch(DATA_URL + "?t=" + Date.now());
      if (!res.ok) throw new Error(res.status);
      serverData = await res.json();
      render();
    } catch {
      document.getElementById("server-list").innerHTML =
        '<div class="empty-state">Could not load data — check back in a moment.<p>Scans run every 3 minutes automatically.</p></div>';
      document.getElementById("status-text").textContent = "Offline";
      document.getElementById("live-dot").style.cssText = "background:#ef4444;box-shadow:0 0 6px #ef4444";
    }
  }

  // ── Utilities ─────────────────────────────────────────────────────────────
  function fmt(sec) {
    const m = Math.floor(sec / 60);
    const s = sec % 60;
    return `${m}m ${String(s).padStart(2, "0")}s`;
  }

  function timeAgo(isoStr) {
    const diff = Math.round((Date.now() - new Date(isoStr)) / 1000);
    if (diff < 60)   return "just now";
    if (diff < 3600) return `${Math.floor(diff / 60)}m ago`;
    return `${Math.floor(diff / 3600)}h ago`;
  }

  // ── Render all ────────────────────────────────────────────────────────────
  function render() {
    if (!serverData) return;

    document.getElementById("stats-row").style.display = "grid";
    document.getElementById("stat-tracked").textContent = serverData.totalTracked;
    document.getElementById("stat-known").textContent   = serverData.totalKnown;
    document.getElementById("stat-new").textContent     = "+" + serverData.newThisScan;
    document.getElementById("stat-died").textContent    = "-" + (serverData.diedThisScan ?? 0);

    document.getElementById("last-update").textContent =
      `Last scan: ${timeAgo(serverData.updatedAt)} · scans every 3 min`;

    document.getElementById("live-dot").style.cssText = "background:#22c55e;box-shadow:0 0 6px #22c55e";
    document.getElementById("status-text").textContent = `${serverData.totalTracked} tracked`;

    document.getElementById("warmup-banner").style.display =
      serverData.totalTracked < 5 ? "block" : "none";

    renderServers();
  }

  // ── Render server cards ───────────────────────────────────────────────────
  function renderServers() {
    if (!serverData) return;
    if (tickInterval) { clearInterval(tickInterval); tickInterval = null; }

    const servers = serverData.events?.[activeEvent] ?? [];
    const list    = document.getElementById("server-list");

    if (servers.length === 0) {
      list.innerHTML = `<div class="empty-state">
        ${serverData.totalTracked < 5
          ? "Still warming up — come back in ~3 minutes once a baseline is built."
          : "No tracked servers for this event yet. Try another tab."}
        <p>Dead servers and servers over 4 hours old are removed automatically.</p>
      </div>`;
      return;
    }

    const capturedAt  = Date.now();
    const baseServers = servers.map(s => ({ ...s }));

    function buildCards() {
      return baseServers.map((s, i) => {
        const elapsed   = Math.round((Date.now() - capturedAt) / 1000);
        const remaining = Math.max(0, s.timeUntilEvent - elapsed);
        const urgent    = remaining < 300;
        const soon      = !urgent && remaining < 600;
        const cls       = urgent ? " urgent" : soon ? " soon" : "";

        return `
          <div class="server-card${cls}">
            <div class="rank-badge${cls}">${i + 1}</div>
            <div class="timer-block">
              <div class="countdown${cls}">${fmt(remaining)}</div>
              <div class="event-sub">until event</div>
            </div>
            <div class="job-id" title="${s.jobId}">${s.jobId}</div>
            <div class="meta-row">
              <div class="meta-item"><span class="val">${s.playing}</span><span class="lbl">players</span></div>
              <div class="meta-item"><span class="val">${s.fps}</span><span class="lbl">fps</span></div>
              <div class="meta-item"><span class="val">${Math.round(s.estimatedAge / 60)}m</span><span class="lbl">age</span></div>
            </div>
            <button class="copy-btn" onclick="copyId('${s.jobId}', this)">Copy ID</button>
          </div>`;
      }).join("");
    }

    list.innerHTML = buildCards();
    tickInterval = setInterval(() => { list.innerHTML = buildCards(); }, 1000);
  }

  // ── Copy ─────────────────────────────────────────────────────────────────
  function copyId(jobId, btn) {
    navigator.clipboard.writeText(jobId).then(() => {
      btn.textContent = "Copied!";
      btn.classList.add("copied");
      setTimeout(() => { btn.textContent = "Copy ID"; btn.classList.remove("copied"); }, 2000);
    });
  }

  // ── Init ─────────────────────────────────────────────────────────────────
  fetchData();
  setInterval(fetchData, REFRESH_MS);
</script>
